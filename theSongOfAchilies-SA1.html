Next import the Data

# import the book
df <- read_csv("~/theSongOfAchilies.txt")
# mutate and get the chapter
df_tidy <- df %>% mutate(
  linenumber = row_number(),
  chapter = cumsum(str_detect(text,
                              regex("^chapter [\\divxlc]", 
                                    ignore_case = TRUE)))) %>% 
  ungroup() %>%
  unnest_tokens(word, text)
# remove stop words
df_tidy <- df_tidy %>% 
  anti_join(stop_words)
# check against nrc
nrc_joy <- get_sentiments("nrc") %>% 
  filter(sentiment == "joy")
# Get the joy sentiment word
df_tidy %>%
  inner_join(nrc_joy) %>%
  count(word, sort = TRUE)

# Join bind lexicon to the tidied df
# with positive and negative.
soa <- df_tidy %>%
  inner_join(get_sentiments("bing")) %>%
  count(index = linenumber %/% 80, sentiment) %>%
  pivot_wider(names_from = sentiment, values_from = n, values_fill = 0) %>% 
  mutate(sentiment = positive - negative)
# Make a plot of the positive vs negative
ggplot(soa, aes(index, sentiment, fill = sentiment)) +
  geom_col(show.legend = F) +
  labs(
    x = "Index",
    y = "Sentiment",
    title = "Sentiment Analysis | bing",
    subtitle = "Book: The Song of Achilies"
  )

# No wonder I cried..
# Compare the lexicons for the book: afinn, bind, nrc
afinn <- df_tidy %>% 
  inner_join(get_sentiments("afinn")) %>% 
  group_by(index = linenumber %/% 80) %>% 
  summarise(sentiment = sum(value)) %>% 
  mutate(method = "AFINN")

bing_and_nrc <- bind_rows(
  df_tidy %>% 
    inner_join(get_sentiments("bing")) %>%
    mutate(method = "Bing et al."),
  df_tidy %>% 
    inner_join(get_sentiments("nrc") %>% 
                 filter(sentiment %in% c("positive", 
                                         "negative"))
    ) %>%
    mutate(method = "NRC")) %>%
  count(method, index = linenumber %/% 80, sentiment) %>%
  pivot_wider(names_from = sentiment,
              values_from = n,
              values_fill = 0) %>% 
  mutate(sentiment = positive - negative)
# bind and plot
bind_rows(afinn, 
bing_and_nrc) %>%
  ggplot(aes(index, sentiment, fill = method)) +
  geom_col(show.legend = FALSE) +
  facet_wrap(~method, ncol = 1, scales = "free_y") +
  labs(
    x = "Page",
    y = "Index",
    title = "Sentiment Analysis",
    subtitle = "Book: The Song of Achilies"
  )

# get positive vs negative
bing_word_counts <- df_tidy %>%
  inner_join(get_sentiments("bing")) %>%
  count(word, sentiment, sort = TRUE) %>%
  ungroup()
# plot the differences
bing_word_counts %>%
  group_by(sentiment) %>%
  slice_max(n, n = 15) %>% 
  ungroup() %>%
  mutate(word = reorder(word, n)) %>%
  ggplot(aes(n, word, fill = sentiment)) +
  geom_col(show.legend = FALSE) +
  facet_wrap(~sentiment, scales = "free_y") +
  labs(x = "Contribution to sentiment",
       y = NULL)
